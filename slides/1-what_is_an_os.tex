        \item IEEE 802.15.4 (a.k.a. ZigBee): $<$250 kbit/s
        \item IEEE 802.16 (a.k.a. Wi-Max): $<$40 Mbit/s
        \item IEEE 1394 (a.k.a. Firewire): $<$3200 Mbit/s
        \item USB, serial port such as RS-232...
      \end{itemize}
  \end{frame}
